# Standalone integrateStepFunc extracted from the starvz R package.
# Source this file to make integrateStepFunc available in your session:
#   source("integrateStepFunc.R")
#
# Arguments:
#   x      - NumericVector of breakpoints defining the integration intervals
#             (length n+1 for n intervals)
#   bounds - NumericVector of step-function breakpoints
#   values - NumericVector of step-function values between consecutive bounds
#
# Returns:
#   NumericVector of length n with the integral of the step function over
#   each interval [x[i], x[i+1]].

if (!requireNamespace("Rcpp", quietly = TRUE)) {
  stop("Package 'Rcpp' is required. Install it with: install.packages('Rcpp')")
}

Rcpp::cppFunction('
NumericVector integrateStepFunc(NumericVector x, NumericVector bounds, NumericVector values) {
  int nx = x.size() - 1;
  int ix = 0, ib = 0;
  double xcur;
  NumericVector integ(nx);
  // skip bounds that end before the first x interval starts
  for (; bounds[ib + 1] < x[ix]; ib++) {}; ib++;
  for (; ix < nx; ix++) {
    xcur = x[ix];
    double intd = 0;
    for (; bounds[ib] <= x[ix + 1]; ib++) {
      intd += values[ib - 1] * (bounds[ib] - xcur);
      xcur = bounds[ib];
    }
    intd += values[ib - 1] * (x[ix + 1] - xcur);
    xcur = x[ix + 1];
    integ[ix] = intd;
  }
  return integ;
}
')

integrate.stepfunc.Rcpp <- function(inter, b, f) integrateStepFunc(inter, c(-Inf, b, Inf), c(0, f, 0))

getBreaks <- function(dfv = NULL) {
  if (is.null(dfv)) {
    return(NULL)
  }
  breaks <- c(dfv$Start, dfv$End[length(dfv$End)])
  return(breaks)
}

getSlices <- function(dfv = NULL, step = 100) {
  tstart <- dfv %>%
    .$Start %>%
    min()
  tend <- (dfv %>% .$End %>% max())
  # TODO: Explain how we ignore the last slice if tend is not multiple
  # of step. This will make us ignore all behavior from the beginning
  # of that last slice until tend. This is unimportant for visualization
  # purposes but it can be important for stats. On that case, replace
  # the second argument by =tend+step= to make sure all data is considered.
  slices <- c(seq(0, tend, step), tend)
  return(slices)
}

remyTimeIntegration <- function(dfv = NULL, slices = NULL) {
  if (is.null(dfv)) {
    return(NULL)
  }
  if (is.null(slices)) {
    return(NULL)
  }

  # Define breaks
  breaks <- getBreaks(dfv)

  # Define values on breaks
  values <- dfv$Value

  result <- integrate.stepfunc.Rcpp(slices, breaks, values)

  return(result)
}

remyTimeIntegrationPrep <- function(dfv = NULL, myStep = 100) {
  if (is.null(dfv)) {
    return(NULL)
  }
  if (nrow(dfv) == 0) {
    return(NULL)
  }
  mySlices <- getSlices(dfv, step = myStep)
  tibble(Slice = mySlices, Value = c(remyTimeIntegration(dfv, slices = mySlices), 0) / myStep)
}

do_the_aggregation <- function(df, step_width) {
  df |>
  select(ResourceId = Container, Task = Value, Start, End) |>
    arrange(ResourceId, Task, Start) |>
    group_by(ResourceId, Task) |>
    reframe(
      bind_rows(
        tibble(
          Start = Start,
          End   = End,
          Active = 1
        ),
        tibble(
          Start = End,
          End   = lead(Start),
          Active = 0
        ) |> filter(!is.na(End)),
        tibble(
          Start = 0,
          End   = first(Start),
          Active = 0
        )
      )
    ) |>
    rename(Value = Active) |>
    arrange(ResourceId, Task, Start) |>
    group_by(ResourceId, Task) |>
    do(remyTimeIntegrationPrep(., myStep = step_width)) |>
    mutate(Start = Slice,
           End = lead(Slice),
           Duration = End - Start) |>
    ungroup() |>
    na.omit() 
}

do_the_reference_aggregation <- function(df, nb_intervals) {
  df |> pull(Start) |> min() -> t.start
  df |> pull(End) |> max() -> t.end
  step_width = (t.end-t.start)/nb_intervals
  df |>
    do_the_aggregation(step_width) |>
    # make it has the same columns for our do_the_viz fun
    mutate(Slice = round(Slice / step_width)) |>
    rename(Container = ResourceId,
           Value = Task,
           Percent = Value,
           StartTS = Start, # Start of the timeslice (TS)
           EndTS = End,     # End of the timeslice (TS)
           DurationTS = Duration) |> # Duration of the timeslice (TS)
    mutate(Time.Sum = Percent * DurationTS)
}

library(patchwork)
do_the_viz <- function(df, has.summary = FALSE)
{
  df |>
    rename(Rank = Container) -> df
  
  # Prep Y labels
  df |>
    distinct(Rank) |>
    arrange(Rank) |>
    mutate(Position = 0:(n()-1)) |>
    mutate(Label = Position) -> yconfm
  
  df |>
    filter(Time.Sum != 0) |>
    group_by(Rank, Slice, StartTS, EndTS, DurationTS, Value) |>
    summarize(Time.Sum = sum(Time.Sum), .groups="drop_last") |>
    ungroup(Value) |>
    arrange(Value) |>
    mutate(TaskHeight = Time.Sum / DurationTS) |>
    mutate(TaskPosition = cumsum(TaskHeight) - TaskHeight) |>
    left_join(yconfm, by="Rank") |>
    ggplot() +
    scale_y_continuous(
      breaks = yconfm$Position,
      labels = yconfm$Label
    ) +    
    geom_rect(alpha=.5, aes(
      fill = Value,
      xmin = StartTS,
      xmax = EndTS,
      ymin = Position + TaskPosition,
      ymax = Position + (TaskPosition + TaskHeight)
    )) +
    my_style() +
    theme(axis.title.x = element_blank(),
          axis.text.x = element_blank()) +
    ylab("Rank [index]") +
    xlab("Time [seconds]") -> p

# summary plot
  df |>
    filter(Time.Sum != 0) |>
    group_by(Slice, StartTS, EndTS, DurationTS, Value) |>
    summarize(N=n(), Time.Sum = sum(Time.Sum), .groups="drop_last") |>
    ungroup(Value) |>
    arrange(Value) |>
    mutate(TaskHeight = Time.Sum / (DurationTS * N)) |>
    mutate(TaskPosition = cumsum(TaskHeight) - TaskHeight) |>
    ggplot() +
    coord_cartesian(ylim=c(0,1)) +
    geom_rect(alpha=.5, aes(
      fill = Value,
      xmin = StartTS,
      xmax = EndTS,
      ymin = TaskPosition,
      ymax = (TaskPosition + TaskHeight)
    )) +
    my_style() +
    theme(legend.position="none") +
    ylab("Load [%]") +
    xlab("Time [seconds]") -> p.summary
 
  if (has.summary) {
    p / p.summary + plot_layout(heights = c(1, 0.15))
  }else{
    p
  }
}

our_read_function <- function(arq) {
  linhas <- read_lines(arq, progress = FALSE)
  linhas_state <- linhas[str_starts(linhas, "State")]
  read_csv(I(linhas_state),
           show_col_types=FALSE,
           progress=FALSE,
           col_names = c("Nature",
                         "Container",    
                         "Type",
                         "Start",
                         "End",
                         "Duration",
                         "Imbrication",
                         "Value")) |> # name of the Operation
    mutate(Container = as.integer(gsub("rank-?", "", Container))) |>
    mutate(Value = str_trim(gsub("^\\s*PMPI_", "MPI_", str_trim(Value)))) |>    
    mutate(Value = as.factor(Value)) |>
    filter(Value %in% c("MPI_Irecv", "MPI_Isend", "MPI_Waitall")) |>
    mutate(Value = factor(Value, levels=c("MPI_Irecv", "MPI_Isend", "MPI_Waitall"))) -> df
# Keep only the part that interest us
# Highly dependent on parallel design
  core_start <- df |>    
    filter(Value == "MPI_Irecv") |>    
    pull(Start) |>    
    min()    
  core_end <- df |>    
    filter(Value == "MPI_Waitall") |>    
    pull(End) |>    
    max()    
  df <- df |>    
    filter(Start < core_end) |>    
    filter(End > core_start) |>    
    mutate(Start = pmax(Start, core_start), End = pmin(End, core_end), Duration = End - Start)
# Return the data frame
  return(df)
}

add_compute_state <- function(df) {

  compute <- df |>
    arrange(Container, Start) |>
    group_by(Container) |>
    mutate(next_start = lead(Start)) |>
    filter(!is.na(next_start), next_start > End) |>
    transmute(
      Nature = "State",
      Container,
      Type = "MPI_STATE",
      Start = End,
      End = next_start,
      Duration = End - Start,
      Imbrication = 0,
      Value = factor("Compute", levels = c(levels(df$Value), "Compute"))
    )

  bind_rows(df, compute) |>
    arrange(Container, Start) |>
    ungroup()
}

summary_per_operation <- function(dfa) {
  dfa |>
    group_by(Value) |>
    summarise(
      n      = n(),
      mean   = mean(Duration, na.rm = TRUE),
      sd     = sd(Duration, na.rm = TRUE),
      min    = min(Duration, na.rm = TRUE),
      p25    = quantile(Duration, 0.25, na.rm = TRUE),
      median = median(Duration, na.rm = TRUE),
      p75    = quantile(Duration, 0.75, na.rm = TRUE),
      p90    = quantile(Duration, 0.90, na.rm = TRUE),
      p99    = quantile(Duration, 0.99, na.rm = TRUE),
      max    = max(Duration, na.rm = TRUE),
      .groups = "drop"
    )
}

#+RESULTS:

Since we identified that Compute takes roughly ~80% of time. My
hypothesis indicate that are not the differences of MPI operations
that explain the difference between reality and simulation. It is the
compute. Indeed, when the compute of reality takes place, there is a
lot of concurrency involved (competition). While in simulation, the
compute is alone in the machine, so it gets faster.
